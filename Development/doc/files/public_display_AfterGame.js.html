'use strict';

/**
 *
 * @module display
 * @namespace display
 *
 * @class AfterGame
 * @constructor
 */
function AfterGame(){
	/**
	 * @property _elementId
	 * @type {String}
	 * @default "afterGame"
	 * @private
	 */
	this._elementId = "afterGame";
	
	/**
	 * @property _element
	 * @type {$}
	 * @default null
	 * @private
	 */
	this._element = null;
	
	/**
	 * @property _data
	 * @type {JSON}
	 * @default null
	 * @private
	 */
	this._data = null;
}

/**
 * @method init
 */
AfterGame.prototype.init = function(){
	this._element = $("#"+this._elementId);
	$(window).on("resize", this.reshape.bind(this));
};

/**
 * @method show
 *
 * @param {JSON} data
 */
AfterGame.prototype.show = function(data){
	if(data===null){
		this._element.html("Leider keine Daten verf√ºgbar").show();
		return;
	}
	
	data.spieler.sort(function(a,b){
		return a.platz-b.platz;
	});

	var statistics = data.statistics;
	var spieler = data.spieler;
	var c;
	
	var spielZeitInMs = statistics.gameEnds;
	
	this._element.text("");
	
	$("body").animate({opacity:0}, 550, function(){
		$("#display").text("");
		$("body").css("opacity",1);
	
		this._element.show().css("opacity",0).animate({opacity:1}, 550, function(){
			
			var liste = "<h1>Und der Gewinner lautet: </h1>";
			liste += ("<ul id='gewinnerListe'>");
			for(var i in spieler){
				c = spieler[i].color;
			
				liste += "<li style='border-right-color: rgb("+c[0]+","+c[1]+","+c[2]+")'>";
				
				liste += spieler[i].platz+". "+spieler[i].name;
				
				liste += "<span class='gewinnerListePunkte'>";
				liste += spieler[i].punkte+" Punkte";
				liste += "</span>";
				liste += "<span class='gewinnerListeStuff'>";
				liste += "Level: "+spieler[i].level+"<br />";
				liste += "Lines: "+spieler[i].lines+"";
				liste += "</span>";
				
				liste += "</li>";
			}
			liste += ("</ul>");
			liste += "<h2>Was wann geschah:</h2>";
			liste += "<div id='graph'></div>";
			
			this._element.html(liste);
			
			this.drawGraph(data);
			this._data = data;
			
		}.bind(this));
	}.bind(this));
};

/**
 * @method drawGraph
 *
 * @param {JSON} data
 */
AfterGame.prototype.drawGraph = function(data){
	$("#graph svg").remove();
	
	var statistics = data.statistics;
	var spieler = data.spieler;
	var height = 200;
   	var width = $("#graph").width();
	var end = statistics.gameEnds;
	var elements = [];
	var maxPunkte = 0;
	
	var z, x, y, i ,j, c;
	
	// ermittle die maximal vorkommende punktzahl
	for(i in spieler){
		for(j in statistics.spieler[spieler[i].id].punkteAt){	
			if(maxPunkte < statistics.spieler[spieler[i].id].punkteAt[j]){
				maxPunkte = statistics.spieler[spieler[i].id].punkteAt[j];
			}	
		}
	} 
	
    var paper = Raphael("graph", "100%", height);
    var maxHeight = height-5;
    var path, pathDied;
    
    // horizontale gestrichelte 25% linien.
    for(z = 0; z <= 4; z++){
    	y = Math.round(height-5-(((height-10)/4)*z));
		path = "M0 "+y;
		path += "L"+width+" "+y;
	    c = paper.path(path);
	    c.attr("stroke", "rgba(0,0,0,0.2)");
	    if(z%2==0){
	    	c.attr("stroke-width", "1.5");
	    }else{
	    	c.attr("stroke-width", "1");
	    }
	    c.attr("stroke-dasharray", ".");
    }
    
    // graphen
    var lastx, lasty;
	for(i in spieler){
		path = "M0 "+height;
		pathDied = "";
		
		// ermittle graphpath
		for(j in statistics.spieler[spieler[i].id].punkteAt){
			if(isNaN(j)){
				continue;
			}
			
			x = ((width/end)*j);
			y = height-(maxHeight/maxPunkte)*statistics.spieler[spieler[i].id].punkteAt[j];
			
			if(statistics.spieler[spieler[i].id].died<j){
				if(pathDied===""){
					pathDied = "M"+lastx+" "+lasty;
				}
				pathDied += "L"+x+" "+y;
				
			}else{
				path += "L"+x+" "+(y);
			}
			
			lastx = x;
			lasty = y;
			
		}
		
		// punkteverlauf anzeigen
	    c = paper.path(path);
	    c.attr("stroke", "rgba("+spieler[i].color[0]+","+spieler[i].color[1]+","+spieler[i].color[2]+", 0.9)");
	    c.attr("stroke-width", "3");
	    
	    // nach dem tod weiter seine punkte anzeigen, aber getrichelt
	    c = paper.path(pathDied);
	    c.attr("stroke-width", "1");
	    c.attr("stroke", "rgba("+spieler[i].color[0]+","+spieler[i].color[1]+","+spieler[i].color[2]+", 0.5)");
	    c.attr("stroke-dasharray", "-");
	    
	    // dann noch horizontal bis zum rand
		c = paper.path("M"+x+" "+Math.round(y)+" L"+width+" "+Math.round(y));
	    c.attr("stroke-width", "1");
	    c.attr("stroke", "rgba("+spieler[i].color[0]+","+spieler[i].color[1]+","+spieler[i].color[2]+", 0.5)");
	    c.attr("stroke-dasharray", "-");
	}
	
	// todeszeitpunkt anzeigen
	for(i in spieler){
		j = statistics.spieler[spieler[i].id].died;
		y = height-(maxHeight/maxPunkte)*statistics.spieler[spieler[i].id].punkte;
		x = ((width/end)*j);
		
		// kreis um sein todeszeitpunkt bei punkte y 
		c = paper.circle(x, y, 3);
	    c.attr("fill", "rgba("+spieler[i].color[0]+","+spieler[i].color[1]+","+spieler[i].color[2]+", 0.4)");
	    c.attr("stroke", "rgba("+spieler[i].color[0]+","+spieler[i].color[1]+","+spieler[i].color[2]+", 1)");
	    c.attr("stroke-width", "1");
	    
	    // eine vertikale linie durch sein todeszeitpunkt ziehen
		c = paper.rect(Math.round(x-1), 5, 1, height-10);
	    c.attr("stroke-width", "0");
	    c.attr("fill", "rgba("+spieler[i].color[0]+","+spieler[i].color[1]+","+spieler[i].color[2]+", 0.7)");
	}
	
};

/**
 * @method reshape
 */
AfterGame.prototype.reshape = function(){
	if(this._data !== null){
		this.drawGraph(this._data);
	}
};

/**
 * @method hide
 */
AfterGame.prototype.hide = function(){
	this._element.hide();
};
